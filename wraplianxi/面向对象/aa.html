内部对象按照使用方式分为动态对象和静态对象
	var Opp={//静态对象，直接可以对象名.成员
		name:"小明",
		age:"13",
		sex:"man",
		fn:function(){
			alert(3)
		}
	}
	console.log(Opp.name)
	Opp.fn()


	function fn(){//动态对象
		yu=function(){
			alert(1)
		}
		return yu()
	}
	fn()
	function Fn(){//动态对象 用new实例化之
	后，对象实例名.成员
		this.xi=function(){
			alert(2)
		}
	}
	var f=new Fn();
	f.xi()
